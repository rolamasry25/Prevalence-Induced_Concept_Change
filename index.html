let trialIndex = 0; let responses = []; const totalTrials = 60; const baselineCount = 30; const lowPrevCount = 30; const stimuli = []; function generateColor(t) { const r = Math.round(0 + t * (128 - 0)); const g = Math.round(128 - t * (128 - 0)); const b = 255; return `rgb(${r},${g},${b})`; } for (let i = 0; i < baselineCount; i++) { const t = Math.random(); stimuli.push({ t, prevalence: 'baseline', color: generateColor(t) }); } for (let i = 0; i < lowPrevCount; i++) { const t = Math.random(); stimuli.push({ t, prevalence: 'low', color: generateColor(t) }); } // Shuffle the trials stimuli.sort(() => Math.random() - 0.5); function renderTrial() { const trial = stimuli[trialIndex]; document.getElementById('experiment').innerHTML = `

Trial ${trialIndex + 1} of ${stimuli.length}

`; } window.recordResponse = function (response) { responses.push({ ...stimuli[trialIndex], response }); trialIndex++; if (trialIndex < stimuli.length) { renderTrial(); } else { showSummary(); } }; function showSummary() { document.getElementById('experiment').innerHTML = `

Experiment Complete

You completed ${stimuli.length} trials.

`; } window.downloadCSV = function () { const headers = Object.keys(responses[0]); const csv = [headers.join(',')].concat( responses.map(obj => headers.map(h => obj[h]).join(',')) ).join('\n'); const blob = new Blob([csv], { type: 'text/csv' }); const url = URL.createObjectURL(blob); const a = document.createElement('a'); a.href = url; a.download = 'experiment_results.csv'; a.click(); }; renderTrial();
